def each
  if @size > 0

    curr = @head.next

    begin
      yield curr
      curr = curr.next
    end while(curr != nil)
  end
end

def remove(val)
  # iterate through linkedlist until you find the val, then remove it
  curr = @head
  prev = curr
  while(curr.val != val)
    prev = curr
    curr = curr.next
  end
  prev.set_next(curr.next)
  curr = nil
  @curr = prev
  @size -= 1
  val
end
